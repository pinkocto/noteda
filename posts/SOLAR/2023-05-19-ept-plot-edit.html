%%R
ept = function(y){
    # EpM = eptransf(signal=y,tau=24,process=c("envelope","average"))$EpM
    # EpM*2
    Ep = eptransf(signal=y,tau=24,process=c("envelope","average"), type='rectangle')
    Ep$EpU + Ep$EpL ## EPU
}
